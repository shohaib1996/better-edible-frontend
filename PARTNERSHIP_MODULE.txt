ANSWERED DESIGN DECISIONS

- Billing is MANUAL — admin clicks "Generate Bill" per store per month
- Driver reconciliation happens through the admin panel (no separate driver app)

--- FILE: src/models/PartnershipEnrollment.ts ---

Interface IPartnershipEnrollment:
  storeId          : Types.ObjectId  (ref: "Store", required, unique)
  status           : "pending_approval" | "active" | "pending_setup" | "rejected" | "not_enrolled"
  posApiKey        : string (optional — generated on approval)
  posApiConnected  : boolean (default: false — set true on first successful POS call)
  posApiSku        : Map or leave SKU on inventory level (see note below)
  approvedAt       : Date (optional)
  approvedBy       : string (optional — admin name)
  requestedAt      : Date (default: Date.now)
  notes            : string (optional — admin internal notes)
  createdAt, updatedAt (timestamps: true)

Index: { storeId: 1 } unique

NOTE: posApiKey is generated with crypto.randomBytes(32).toString("hex") on approval.


--- FILE: src/models/PartnershipInventory.ts ---

Interface IPartnershipInventory:
  storeId          : Types.ObjectId  (ref: "Store", required)
  productId        : Types.ObjectId  (ref: "Product", required)
  sku              : string  (required — the SKU the POS will send, e.g. "BE-GUMMY-250")
  productName      : string  (denormalized — copy from Product at time of placement)
  wholesalePrice   : number  (price per unit — used for billing)
  unitsPlaced      : number  (default: 0 — cumulative total ever placed)
  unitsSold        : number  (default: 0 — cumulative total sold, driven by POS)
  unitsRemaining   : number  (default: 0 — updated by POS sales and driver reconciliation)
  lastReconciliationAt : Date (optional — when driver last physically counted)
  createdAt, updatedAt (timestamps: true)

Index: { storeId: 1, productId: 1 } unique
Index: { storeId: 1, sku: 1 } unique  ← used by POS API lookup


--- FILE: src/models/PartnershipSale.ts ---

Interface IPartnershipSale:
  storeId          : Types.ObjectId  (ref: "Store", required)
  productId        : Types.ObjectId  (ref: "Product", required)
  sku              : string  (the SKU from POS payload — store for audit)
  date             : Date  (required — the calendar date of the sale, midnight UTC)
  unitsSold        : number  (required, min: 0)
  source           : "pos_api" (always — no manual entry)
  receivedAt       : Date  (default: Date.now — when our server received it)
  createdAt, updatedAt (timestamps: true)

Index: { storeId: 1, productId: 1, date: 1 } unique  ← idempotent upserts


--- FILE: src/models/PartnershipReplenishment.ts ---

Interface IReplenishmentItem:
  productId        : Types.ObjectId
  productName      : string
  sku              : string
  unitsRequested   : number
  unitsDelivered   : number (optional — filled on delivery)

Interface IDriverCount:
  productId        : Types.ObjectId
  sku              : string
  actualCount      : number  (physical count by driver — overrides unitsRemaining)

Interface IPartnershipReplenishment:
  storeId          : Types.ObjectId  (ref: "Store", required)
  status           : "pending" | "in_transit" | "delivered" | "reconciled"
  items            : IReplenishmentItem[]
  driverCounts     : IDriverCount[]  (default: [])
  driverNotes      : string (optional)
  requestedAt      : Date (default: Date.now)
  deliveredAt      : Date (optional)
  createdAt, updatedAt (timestamps: true)

Index: { storeId: 1, status: 1 }


--- FILE: src/models/PartnershipBill.ts ---

Interface IBillLineItem:
  productId        : Types.ObjectId
  productName      : string
  sku              : string
  unitsSold        : number
  wholesalePrice   : number
  lineTotal        : number  (unitsSold × wholesalePrice)

Interface IBillCredit:
  amount           : number
  reason           : string
  appliedAt        : Date

Interface IPartnershipBill:
  storeId          : Types.ObjectId  (ref: "Store", required)
  billingYear      : number  (e.g. 2026)
  billingMonth     : number  (1–12)
  lineItems        : IBillLineItem[]
  subtotal         : number
  credits          : IBillCredit[]  (default: [])
  creditsTotal     : number  (sum of all credits)
  total            : number  (subtotal - creditsTotal)
  status           : "draft" | "sent" | "paid"
  generatedAt      : Date (default: Date.now)
  paidAt           : Date (optional)
  createdAt, updatedAt (timestamps: true)

Index: { storeId: 1, billingYear: 1, billingMonth: 1 } unique

STEP 2 — CREATE VALIDATORS
--------------------------------------------------------------------------------
Location: src/validators/partnershipSchemas.ts

Schemas needed (Zod):

joinPartnershipSchema:
  (no body fields — uses auth to identify store)

approvePartnershipSchema:
  notes: z.string().optional()

rejectPartnershipSchema:
  notes: z.string().optional()

placeInventorySchema:
  productId        : z.string().min(1)
  sku              : z.string().min(1).max(50)  ← trim and uppercase recommended
  wholesalePrice   : z.number().min(0)
  unitsToAdd       : z.number().int().min(1)

createReplenishmentSchema:
  items: z.array(z.object({
    productId      : z.string().min(1)
    unitsRequested : z.number().int().min(1)
  })).min(1)

deliverReplenishmentSchema:
  driverCounts: z.array(z.object({
    productId    : z.string().min(1)
    actualCount  : z.number().int().min(0)
  })).min(1)
  driverNotes  : z.string().optional()

posInboundSalesSchema:
  date: z.string().regex(/^\d{4}-\d{2}-\d{2}$/)  ← "YYYY-MM-DD"
  items: z.array(z.object({
    sku          : z.string().min(1)
    unitsSold    : z.number().int().min(0)
  })).min(1)

generateBillSchema:
  year  : z.number().int().min(2024)
  month : z.number().int().min(1).max(12)

applyCreditSchema:
  amount  : z.number().min(0.01)
  reason  : z.string().min(1)

updateBillStatusSchema:
  status: z.enum(["draft", "sent", "paid"])

STEP 3 — CREATE CONTROLLERS
--------------------------------------------------------------------------------
Location: src/controllers/partnership/

Split into focused files:

--- partnershipEnrollmentController.ts ---
  joinPartnership(req, res)
    - Check if enrollment already exists for storeId (from auth)
    - If exists and status is not "rejected" → error "Already enrolled or pending"
    - Create PartnershipEnrollment with status "pending_approval"
    - Return 201

  getPartnershipStatus(req, res)
    - Find PartnershipEnrollment by storeId
    - If not found → return { status: "not_enrolled" }
    - Return enrollment doc

  approvePartnership(req, res)  [admin]
    - Find enrollment by storeId param
    - Generate posApiKey: crypto.randomBytes(32).toString("hex")
    - Set status = "pending_setup", posApiKey, approvedAt, approvedBy from admin session
    - Save and return

  rejectPartnership(req, res)  [admin]
    - Find enrollment by storeId param
    - Set status = "rejected", notes from body
    - Save and return

  getAllPartnershipStores(req, res)  [admin]
    - Find all PartnershipEnrollment docs
    - Populate storeId with name, city, state
    - Support query param: status filter
    - Return list


--- partnershipInventoryController.ts ---
  getInventory(req, res)
    - Find all PartnershipInventory by storeId (from auth for store, from param for admin)
    - Return list

  placeInventory(req, res)  [admin]
    - Validate storeId param + body (productId, sku, wholesalePrice, unitsToAdd)
    - Upsert PartnershipInventory: $inc unitsPlaced + unitsRemaining by unitsToAdd
    - On insert: set productName (lookup Product), sku, wholesalePrice
    - Return updated doc

  IMPORTANT: After driver delivers (deliverReplenishment), call a helper
  that sets unitsRemaining = driverCount.actualCount for each product.
  This is the reconciliation override — driver count is the source of truth.


--- partnershipPosController.ts ---
  receivePosData(req, res)
    - Auth: read X-Partnership-Key header
    - Find PartnershipEnrollment by posApiKey
    - If not found or status !== "active" → 401
    - Validate body with posInboundSalesSchema
    - For each item in body.items:
        * Find PartnershipInventory by { storeId, sku }
        * If not found → skip with warning (log it)
        * Parse date to UTC midnight
        * Upsert PartnershipSale (storeId + productId + date) with unitsSold
          Use findOneAndUpdate with $set so re-sends are idempotent
        * After upsert: recalculate unitsSold in inventory
          → aggregate SUM of all PartnershipSale.unitsSold for this storeId+productId
          → set inventory.unitsSold = that sum
          → set inventory.unitsRemaining = inventory.unitsPlaced - inventory.unitsSold
    - Set enrollment.posApiConnected = true (if not already)
    - Return { success: true, processed: N, skipped: M }


--- partnershipReplenishmentController.ts ---
  getReplenishments(req, res)
    - Find PartnershipReplenishment by storeId, sort by requestedAt desc
    - Return list

  createReplenishment(req, res)  [admin]
    - storeId from param, items from body
    - For each item: lookup PartnershipInventory to get productName + sku
    - Create PartnershipReplenishment with status "pending"
    - Return 201

  updateReplenishmentStatus(req, res)  [admin]
    - PATCH status only (pending → in_transit → delivered)
    - If setting delivered: set deliveredAt = now

  deliverReplenishment(req, res)  [admin]
    - body: { driverCounts: [{ productId, actualCount }], driverNotes }
    - Find replenishment, must be in_transit or delivered
    - Set driverCounts, driverNotes, status = "reconciled"
    - For each driverCount:
        * Set PartnershipInventory.unitsRemaining = actualCount  ← OVERRIDE
        * Set PartnershipInventory.lastReconciliationAt = now
    - Return updated replenishment


--- partnershipBillingController.ts ---
  getBilling(req, res)
    - Find PartnershipBill by storeId, sort by year+month desc
    - Return list

  generateBill(req, res)  [admin]
    - storeId from param, { year, month } from body
    - Check no bill already exists for this storeId + year + month
    - Query PartnershipSale: all records for storeId where date falls in that month
    - Group by productId → sum unitsSold
    - For each product: lookup PartnershipInventory for wholesalePrice + productName + sku
    - Build lineItems array
    - Calculate subtotal = sum of lineTotals
    - Create PartnershipBill with status "draft", credits = [], creditsTotal = 0, total = subtotal
    - Return bill

  applyCredit(req, res)  [admin]
    - Find bill by id param
    - Push { amount, reason, appliedAt: now } to credits[]
    - Recalculate creditsTotal = sum of credits
    - Recalculate total = subtotal - creditsTotal
    - Save and return

  updateBillStatus(req, res)  [admin]
    - PATCH status: "draft" → "sent" → "paid"
    - If setting "paid": set paidAt = now
    - Save and return

    provides: ["partnershipStores"]

  useApprovePartnershipMutation()
    POST /admin/partnership/:storeId/approve
    invalidates: ["partnershipStores"]

  useRejectPartnershipMutation()
    POST /admin/partnership/:storeId/reject
    invalidates: ["partnershipStores"]

  useGetAdminInventoryQuery(storeId: string)
    GET /admin/partnership/:storeId/inventory

  usePlaceInventoryMutation()
    POST /admin/partnership/:storeId/inventory
    body: { productId, sku, wholesalePrice, unitsToAdd }
    invalidates: ["partnershipInventory"]

  useGetAdminSalesQuery(storeId: string, params)
    GET /admin/partnership/:storeId/sales

  useGetAdminReplenishmentsQuery(storeId: string)
    GET /admin/partnership/:storeId/replenishments

  useCreateReplenishmentMutation()

  useUpdateReplenishmentStatusMutation()

    invalidates: ["partnershipReplenishments"]

  useDeliverReplenishmentMutation()
    POST /admin/partnership/replenishment/:id/deliver

  useGetAdminBillingQuery(storeId: string)
